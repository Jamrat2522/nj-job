-- ═══════════════════════════════════════════════════════
-- N.J. Logistics Billing DB — Schema + Role-based RLS
-- Admin เห็นทุกคน / AC เห็นเฉพาะของตัวเอง
-- รัน 1 ครั้งใน Supabase → SQL Editor → Run
-- ═══════════════════════════════════════════════════════

-- 1. TABLES ---------------------------------------------
create table if not exists public.billing_sessions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade not null,
  user_email text,
  filename text not null,
  file_type text not null check (file_type in ('SER','ADV')),
  target_sheet text not null,
  voucher_count int not null default 0,
  detail_count int not null default 0,
  imported_at timestamptz not null default now()
);
alter table public.billing_sessions add column if not exists user_email text;

create table if not exists public.billing_vouchers (
  id uuid primary key default gen_random_uuid(),
  session_id uuid references public.billing_sessions(id) on delete cascade not null,
  user_id uuid references auth.users(id) on delete cascade not null,
  voucher_no text not null,
  voucher_date text,
  file_type text not null,
  created_at timestamptz not null default now()
);

create table if not exists public.billing_details (
  id uuid primary key default gen_random_uuid(),
  voucher_id uuid references public.billing_vouchers(id) on delete cascade not null,
  user_id uuid references auth.users(id) on delete cascade not null,
  shipment_ref text, ref_type text, charge_code text,
  net_amount numeric, tax_code text, global_code text, remark text,
  row_order int not null default 0,
  created_at timestamptz not null default now()
);

-- 2. USER PROFILES (role: admin | ac) -------------------
create table if not exists public.user_profiles (
  user_id uuid primary key references auth.users(id) on delete cascade,
  email text,
  role text not null default 'ac' check (role in ('admin','ac')),
  created_at timestamptz default now()
);

-- auto-create profile เมื่อมี user ใหม่
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.user_profiles (user_id, email, role)
  values (new.id, new.email, 'ac')
  on conflict (user_id) do nothing;
  return new;
end;
$$;
drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created after insert on auth.users
  for each row execute function public.handle_new_user();

-- backfill users ที่มีอยู่แล้ว
insert into public.user_profiles (user_id, email, role)
select id, email, 'ac' from auth.users
on conflict (user_id) do nothing;

-- 3. is_admin() helper ----------------------------------
create or replace function public.is_admin()
returns boolean language sql security definer stable as $$
  select coalesce((select role='admin' from public.user_profiles where user_id=auth.uid()), false);
$$;

-- 4. RLS ------------------------------------------------
alter table public.billing_sessions enable row level security;
alter table public.billing_vouchers enable row level security;
alter table public.billing_details enable row level security;
alter table public.user_profiles enable row level security;

drop policy if exists "own sessions" on public.billing_sessions;
drop policy if exists "own vouchers" on public.billing_vouchers;
drop policy if exists "own details"  on public.billing_details;
drop policy if exists "read sessions" on public.billing_sessions;
drop policy if exists "ins sessions"  on public.billing_sessions;
drop policy if exists "upd sessions"  on public.billing_sessions;
drop policy if exists "del sessions"  on public.billing_sessions;

create policy "read sessions" on public.billing_sessions for select using (auth.uid()=user_id or public.is_admin());
create policy "ins sessions"  on public.billing_sessions for insert with check (auth.uid()=user_id);
create policy "upd sessions"  on public.billing_sessions for update using (auth.uid()=user_id or public.is_admin());
create policy "del sessions"  on public.billing_sessions for delete using (auth.uid()=user_id or public.is_admin());

drop policy if exists "read vouchers" on public.billing_vouchers;
drop policy if exists "ins vouchers"  on public.billing_vouchers;
drop policy if exists "upd vouchers"  on public.billing_vouchers;
drop policy if exists "del vouchers"  on public.billing_vouchers;
create policy "read vouchers" on public.billing_vouchers for select using (auth.uid()=user_id or public.is_admin());
create policy "ins vouchers"  on public.billing_vouchers for insert with check (auth.uid()=user_id);
create policy "upd vouchers"  on public.billing_vouchers for update using (auth.uid()=user_id or public.is_admin());
create policy "del vouchers"  on public.billing_vouchers for delete using (auth.uid()=user_id or public.is_admin());

drop policy if exists "read details" on public.billing_details;
drop policy if exists "ins details"  on public.billing_details;
drop policy if exists "upd details"  on public.billing_details;
drop policy if exists "del details"  on public.billing_details;
create policy "read details" on public.billing_details for select using (auth.uid()=user_id or public.is_admin());
create policy "ins details"  on public.billing_details for insert with check (auth.uid()=user_id);
create policy "upd details"  on public.billing_details for update using (auth.uid()=user_id or public.is_admin());
create policy "del details"  on public.billing_details for delete using (auth.uid()=user_id or public.is_admin());

drop policy if exists "read profile"   on public.user_profiles;
drop policy if exists "admin upd role" on public.user_profiles;
create policy "read profile"   on public.user_profiles for select using (auth.uid()=user_id or public.is_admin());
create policy "admin upd role" on public.user_profiles for update using (public.is_admin()) with check (public.is_admin());

create index if not exists idx_sessions_user on public.billing_sessions(user_id, imported_at desc);
create index if not exists idx_vouchers_session on public.billing_vouchers(session_id);
create index if not exists idx_details_voucher on public.billing_details(voucher_id);
