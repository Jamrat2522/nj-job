📋 SCHEMA (สร้าง table `profiles` + trigger ใน Supabase SQL Editor) 
-- 1) Profiles table — เก็บ role/name/avatar (mirror auth.users)
create table if not exists profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  username text,
  name text,
  role text not null default 'User',
  avatar text,
  color text default '#f97316',
  created_at timestamptz default now()
);

-- 2) Auto-create profile เมื่อ auth.users มีใหม่
create or replace function handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.profiles (id, username, name, role, avatar, color)
  values (
    new.id,
    split_part(new.email, '@', 1),
    coalesce(new.raw_user_meta_data->>'name', split_part(new.email,'@',1)),
    coalesce(new.raw_user_meta_data->>'role', 'User'),
    coalesce(new.raw_user_meta_data->>'avatar', upper(substr(new.email,1,1))),
    coalesce(new.raw_user_meta_data->>'color', '#f97316')
  );
  return new;
end; $$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute function handle_new_user();

-- 3) RLS: อ่านได้ทุก user ที่ login, แก้/ลบได้เฉพาะ Super Admin
alter table profiles enable row level security;
create policy "read_authed" on profiles for select to authenticated using (true);
create policy "admin_write" on profiles for all to authenticated using (
  exists (select 1 from profiles where id = auth.uid() and role = 'Super Admin')
);

-- 4) อัพเดต Jamrat เป็น Super Admin (หลัง signUp เสร็จ)
update profiles set role = 'Super Admin' where username = 'jamrat';
